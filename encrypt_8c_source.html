125 PublicKey_import(PublicKey* pk, const unsigned char* data, unsigned int dataLen)
126 {
127  SECStatus rv = SECSuccess;
128  CERTSubjectPublicKeyInfo* pkinfo = NULL;
129  *pk = NULL;
130  unsigned char* key_bytes = NULL;
131  uint8_t* spki_data = NULL;
132 
133  if (dataLen != CURVE25519_KEY_LEN)
134  return SECFailure;
135 
136  P_CHECKA(key_bytes = calloc(dataLen, sizeof(unsigned char)));
137  memcpy(key_bytes, data, dataLen);
138 
139  const int spki_len = sizeof(curve25519_spki_zeros);
140  P_CHECKA(spki_data = calloc(spki_len, sizeof(uint8_t)));
141 
142  memcpy(spki_data, curve25519_spki_zeros, spki_len);
143  SECItem spki_item = { siBuffer, spki_data, spki_len };
144 
145  // Import the all-zeros curve25519 public key.
146  P_CHECKA(pkinfo = SECKEY_DecodeDERSubjectPublicKeyInfo(&spki_item));
147  P_CHECKA(*pk = SECKEY_ExtractPublicKey(pkinfo));
148 
149  // Overwrite the all-zeros public key with the 32-byte curve25519 public key
150  // given as input.
151  memcpy((*pk)->u.ec.publicValue.data, data, CURVE25519_KEY_LEN);
152 
153 cleanup:
154  if (key_bytes)
155  free(key_bytes);
156  if (spki_data)
157  free(spki_data);
158  if (pkinfo)
159  SECKEY_DestroySubjectPublicKeyInfo(pkinfo);
160 
161  if (rv != SECSuccess)
162  PublicKey_clear(*pk);
163  return rv;
164 }

300 static void
301 key_to_hex(const unsigned char key_in[CURVE25519_KEY_LEN],
302  unsigned char hex_out[(2 * CURVE25519_KEY_LEN) + 1])
303 {
304  const unsigned char* p = key_in;
305  for (unsigned int i = 0; i < CURVE25519_KEY_LEN; i++) {
306  unsigned char bytel = p[0] & 0x0f;
307  unsigned char byteu = (p[0] & 0xf0) >> 4;
308  hex_out[2 * i] = int_to_hex(byteu);
309  hex_out[2 * i + 1] = int_to_hex(bytel);
310  p++;
311  }
312 
313  hex_out[2 * CURVE25519_KEY_LEN] = '\0';
314 }

316 static SECStatus
317 key_from_hex(unsigned char key_out[CURVE25519_KEY_LEN],
318  const unsigned char hex_in[CURVE25519_KEY_LEN_HEX])
319 {
320  for (unsigned int i = 0; i < CURVE25519_KEY_LEN_HEX; i++) {
321  if (!is_hex_digit(hex_in[i]))
322  return SECFailure;
323  }
324 
325  const unsigned char* p = hex_in;
326  for (unsigned int i = 0; i < CURVE25519_KEY_LEN; i++) {
327  uint8_t d0 = hex_to_int(p[0]);
328  uint8_t d1 = hex_to_int(p[1]);
329  key_out[i] = (d0 << 4) | d1;
330  p += 2;
331  }
332 
333  return SECSuccess;
334 }

471 PublicKey_encrypt(PublicKey pubkey, unsigned char* output,
472  unsigned int* outputLen, unsigned int maxOutputLen,
473  const unsigned char* input, unsigned int inputLen)
474 {

479  return SECFailure;
480 
481  unsigned int needLen;
482  if (PublicKey_encryptSize(inputLen, &needLen) != SECSuccess)
483  return SECFailure;
484 
485  if (maxOutputLen < needLen)
486  return SECFailure;
487 
488  SECStatus rv = SECSuccess;
489  PublicKey eph_pub = NULL;
490  PrivateKey eph_priv = NULL;
491  PK11SymKey* aes_key = NULL;
492 
493  unsigned char nonce[GCM_IV_LEN_BYTES];
494  unsigned char aadBuf[AAD_LEN];
495  P_CHECKC(rand_bytes(nonce, GCM_IV_LEN_BYTES));
496 
497  P_CHECKC(Keypair_new(&eph_priv, &eph_pub));
498  P_CHECKC(derive_dh_secret(&aes_key, eph_priv, pubkey));
499 
500  CK_GCM_PARAMS param;
501  SECItem paramItem;
502  set_gcm_params(&paramItem, &param, nonce, eph_pub, aadBuf);
503

505  P_CHECKCB(pk->len == CURVE25519_KEY_LEN);
506  memcpy(output, pk->data, pk->len);
507  memcpy(output + CURVE25519_KEY_LEN, param.pIv, param.ulIvLen);
508 
509  const int offset = CURVE25519_KEY_LEN + param.ulIvLen;
510  P_CHECKC(PK11_Encrypt(aes_key, CKM_AES_GCM, &paramItem, output + offset,
511  outputLen, maxOutputLen - offset, input, inputLen));
512  *outputLen = *outputLen + CURVE25519_KEY_LEN + GCM_IV_LEN_BYTES;
513 
514 cleanup:
515  PublicKey_clear(eph_pub);
516  PrivateKey_clear(eph_priv);
517  if (aes_key)
518  PK11_FreeSymKey(aes_key);
519 
520  return rv;
521 }
