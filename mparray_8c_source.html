65 {

68 
69  if (oldlen == newlen)
70  return rv;
71 
72  // TODO: Use realloc for this?

74  if (newdata == NULL)
75  return SECFailure;
76 
77  for (int i = 0; i < newlen; i++) {

80

88  MP_CHECKC(mp_copy(&arr->data[i], &newdata[i]));

102  mp_clear(&newdata[i]);
