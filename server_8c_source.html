185 static SECStatus
186 compute_shares(PrioVerifier v, const_PrioPacketClient p)
187 {
188  SECStatus rv;
189  const int n = v->s->cfg->num_data_fields + 1;
190  const int N = next_power_of_two(n);
191  mp_int eval_at;
192  MP_DIGITS(&eval_at) = NULL;
193 
194  MPArray points_f = NULL;
195  MPArray points_g = NULL;
196  MPArray points_h = NULL;
197 
198  MP_CHECKC(mp_init(&eval_at));
199  P_CHECKA(points_f = MPArray_new(N));
200  P_CHECKA(points_g = MPArray_new(N));
201  P_CHECKA(points_h = MPArray_new(2 * N));
202 
203  // Use PRG to generate random point
204  MP_CHECKC(PRG_get_int(v->s->prg, &eval_at, &v->s->cfg->modulus));
205 
206  // Reduce value into the field we're using. This
207  // doesn't yield exactly a uniformly random point,
208  // but for values this large, it will be close
209  // enough.
210  MP_CHECKC(mp_mod(&eval_at, &v->s->cfg->modulus, &eval_at));
211 
212  // Client sends us the values of f(0) and g(0)
213  MP_CHECKC(mp_copy(&p->f0_share, &points_f->data[0]));
214  MP_CHECKC(mp_copy(&p->g0_share, &points_g->data[0]));
215  MP_CHECKC(mp_copy(&p->h0_share, &points_h->data[0]));
216 
217  for (int i = 1; i < n; i++) {
218  // [f](i) = i-th data share
219  const mp_int* data_i_minus_1 = get_data_share(v, i - 1);
220  MP_CHECKC(mp_copy(data_i_minus_1, &points_f->data[i]));
221 
222  // [g](i) = i-th data share minus 1
223  // Only need to shift the share for 0-th server
224  MP_CHECKC(mp_copy(&points_f->data[i], &points_g->data[i]));
225  if (!v->s->idx) {
226  MP_CHECKC(mp_sub_d(&points_g->data[i], 1, &points_g->data[i]));
227  MP_CHECKC(
228  mp_mod(&points_g->data[i], &v->s->cfg->modulus, &points_g->data[i]));
229  }
230  }
231 
232  int j = 0;
233  for (int i = 1; i < 2 * N; i += 2) {
234  const mp_int* h_point_j = get_h_share(v, j++);
235  MP_CHECKC(mp_copy(h_point_j, &points_h->data[i]));
236  }
237 
238  P_CHECKC(poly_interp_evaluate(&v->share_fR, points_f, &eval_at, v->s->cfg));
239  P_CHECKC(poly_interp_evaluate(&v->share_gR, points_g, &eval_at, v->s->cfg));
240  P_CHECKC(poly_interp_evaluate(&v->share_hR, points_h, &eval_at, v->s->cfg));
241 
242 cleanup:
243  MPArray_clear(points_f);
244  MPArray_clear(points_g);
245  MPArray_clear(points_h);
246  mp_clear(&eval_at);
247  return rv;
248 }

288 SECStatus
289 PrioVerifier_set_data(PrioVerifier v, unsigned char* data,
290  unsigned int data_len)
291 {
292  SECStatus rv = SECSuccess;
293  PRG prgB = NULL;
294  P_CHECKC(PrioPacketClient_decrypt(v->clientp, v->s->cfg, v->s->priv_key, data,
295  data_len));
296 
297  PrioPacketClient p = v->clientp;
298  if (p->for_server != v->s->idx)
299  return SECFailure;
300 
301  const int N = next_power_of_two(v->s->cfg->num_data_fields + 1);
302  if (v->s->idx == PRIO_SERVER_A) {
303  // Check that packet has the correct number of data fields
304  if (p->shares.A.data_shares->len != v->s->cfg->num_data_fields)
305  return SECFailure;
306  if (p->shares.A.h_points->len != N)
307  return SECFailure;
308  }
309 
310  if (v->s->idx == PRIO_SERVER_B) {
311  P_CHECKA(prgB = PRG_new(v->clientp->shares.B.seed));
312  P_CHECKC(PRG_get_array(prgB, v->data_sharesB, &v->s->cfg->modulus));
313  P_CHECKC(PRG_get_array(prgB, v->h_pointsB, &v->s->cfg->modulus));
314  }
315 
316  // TODO: This can be done much faster by using the combined
317  // interpolate-and-evaluate optimization described in the
318  // Prio paper.
319  //
320  // Compute share of f(r), g(r), h(r)
321  P_CHECKC(compute_shares(v, p));
322 
323 cleanup:
324 
325  PRG_clear(prgB);
326  return rv;
327 }

424 SECStatus
425 PrioPacketVerify2_set_data(PrioPacketVerify2 p2, const_PrioVerifier v,
426  const_PrioPacketVerify1 p1A,
427  const_PrioPacketVerify1 p1B)
428 {
429  SECStatus rv = SECSuccess;
430 
431  mp_int d, e, tmp;
432  MP_DIGITS(&d) = NULL;
433  MP_DIGITS(&e) = NULL;
434  MP_DIGITS(&tmp) = NULL;
435 
436  MP_CHECKC(mp_init(&d));
437  MP_CHECKC(mp_init(&e));
438  MP_CHECKC(mp_init(&tmp));
439 
440  const mp_int* mod = &v->s->cfg->modulus;
441 
442  // Compute share of f(r)*g(r)
443  // [f(r)*g(r)] = [d*e/2] + d[b] + e[a] + [c]
444 
445  // Compute d
446  MP_CHECKC(mp_addmod(&p1A->share_d, &p1B->share_d, mod, &d));
447  // Compute e
448  MP_CHECKC(mp_addmod(&p1A->share_e, &p1B->share_e, mod, &e));
449 
450  // Compute d*e
451  MP_CHECKC(mp_mulmod(&d, &e, mod, &p2->share_out));
452  // out = d*e/2
453  MP_CHECKC(mp_mulmod(&p2->share_out, &v->s->cfg->inv2, mod, &p2->share_out));
454 
455  // Compute d[b]
456  MP_CHECKC(mp_mulmod(&d, &v->clientp->triple->b, mod, &tmp));
457  // out = d*e/2 + d[b]
458  MP_CHECKC(mp_addmod(&p2->share_out, &tmp, mod, &p2->share_out));
459 
460  // Compute e[a]
461  MP_CHECKC(mp_mulmod(&e, &v->clientp->triple->a, mod, &tmp));
462  // out = d*e/2 + d[b] + e[a]
463  MP_CHECKC(mp_addmod(&p2->share_out, &tmp, mod, &p2->share_out));
464 
465  // out = d*e/2 + d[b] + e[a] + [c]
466  MP_CHECKC(
467  mp_addmod(&p2->share_out, &v->clientp->triple->c, mod, &p2->share_out));
468 
469  // We want to compute f(r)*g(r) - h(r),
470  // so subtract off [h(r)]:
471  // out = d*e/2 + d[b] + e[a] + [c] - [h(r)]
472  MP_CHECKC(mp_sub(&p2->share_out, &v->share_hR, &p2->share_out));
473  MP_CHECKC(mp_mod(&p2->share_out, mod, &p2->share_out));
474 
475 cleanup:
476  mp_clear(&d);
477  mp_clear(&e);
478  mp_clear(&tmp);
479  return rv;
480 }

483 PrioVerifier_isValid(const_PrioVerifier v, const_PrioPacketVerify2 pA,
484  const_PrioPacketVerify2 pB)
485 {

487  mp_int res;

489  MP_CHECKC(mp_init(&res));

492  // ensure that the sum is equal to zero, which indicates

499
