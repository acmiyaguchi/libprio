27  const mp_int* ys, mp_int* tmp, mp_int* ySub, mp_int* rootsSub)
28 {
29  if (n == 1) {
30  MP_CHECK(mp_copy(&ys[0], &out[0]));
31  return SECSuccess;
32  }
33 
34  // Recurse on the first half
35  for (int i = 0; i < n / 2; i++) {
36  MP_CHECK(mp_addmod(&ys[i], &ys[i + (n / 2)], mod, &ySub[i]));
37  MP_CHECK(mp_copy(&roots[2 * i], &rootsSub[i]));
38  }
39 
40  MP_CHECK(fft_recurse(tmp, mod, n / 2, rootsSub, ySub, &tmp[n / 2],
41  &ySub[n / 2], &rootsSub[n / 2]));
42  for (int i = 0; i < n / 2; i++) {
43  MP_CHECK(mp_copy(&tmp[i], &out[2 * i]));
44  }
45 
46  // Recurse on the second half
47  for (int i = 0; i < n / 2; i++) {
48  MP_CHECK(mp_submod(&ys[i], &ys[i + (n / 2)], mod, &ySub[i]));
49  MP_CHECK(mp_mulmod(&ySub[i], &roots[i], mod, &ySub[i]));
50  }
51 
52  MP_CHECK(fft_recurse(tmp, mod, n / 2, rootsSub, ySub, &tmp[n / 2],
53  &ySub[n / 2], &rootsSub[n / 2]));
54  for (int i = 0; i < n / 2; i++) {
55  MP_CHECK(mp_copy(&tmp[i], &out[2 * i + 1]));
56  }
57 
58  return SECSuccess;
59 }

151  const_PrioConfig cfg)
152 {
153  SECStatus rv = SECSuccess;
154  const int n = coeffs->len;
155 
156  // Use Horner's method to evaluate the polynomial at the point
157  // `eval_at`
158  mp_copy(&coeffs->data[n - 1], value);
159  for (int i = n - 2; i >= 0; i--) {
160  MP_CHECK(mp_mulmod(value, eval_at, &cfg->modulus, value));
