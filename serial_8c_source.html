21 serial_write_mp_int(msgpack_packer* pk, const mp_int* n)
22 {
23  SECStatus rv = SECSuccess;
24  unsigned int n_size = mp_unsigned_octet_size(n);
25  unsigned char* data = NULL;
26 
27  P_CHECKA(data = calloc(n_size, sizeof(unsigned char)));
28  MP_CHECKC(mp_to_fixlen_octets(n, data, n_size));
29 
30  P_CHECKC(msgpack_pack_str(pk, n_size));
31  P_CHECKC(msgpack_pack_str_body(pk, data, n_size));
32 cleanup:
33  if (data)

39 object_to_mp_int(msgpack_object* obj, mp_int* n, const mp_int* max)
40 {
41  SECStatus rv = SECSuccess;

46  msgpack_object_str s = obj->via.str;

48  MP_CHECKC(mp_read_unsigned_octets(n, (unsigned char*)s.ptr, s.size));
