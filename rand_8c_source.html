38 rand_bytes_internal(void* user_data, unsigned char* out, size_t n_bytes)
39 {
40  // No pointer should ever be passed in.
41  if (user_data != NULL)
42  return SECFailure;
43  if (!NSS_IsInitialized()) {
44  PRIO_DEBUG("NSS not initialized. Call rand_init() first.");
45  return SECFailure;
46  }
47 
48  SECStatus rv = SECFailure;
49 
50  int to_go = n_bytes;
51  unsigned char* cp = out;
52  while (to_go) {
53  int to_gen = MIN(CHUNK_SIZE, to_go);
54  if ((rv = PK11_GenerateRandom(cp, to_gen)) != SECSuccess) {
55  PRIO_DEBUG("Error calling PK11_GenerateRandom");
56  return SECFailure;
57  }
58 
59  cp += CHUNK_SIZE;
60  to_go -= to_gen;
61  }
62 
63  return rv;
64 }

78 SECStatus
79 rand_int_rng(mp_int* out, const mp_int* max, RandBytesFunc rng_func,
80  void* user_data)
81 {
82  SECStatus rv = SECSuccess;
83  unsigned char* max_bytes = NULL;
84  unsigned char* buf = NULL;
85 
86  // Ensure max value is > 0
87  if (mp_cmp_z(max) == 0)
88  return SECFailure;
89 
90  // Compute max-1, which tells us the largest
91  // value we will ever need to generate.
92  MP_CHECKC(mp_sub_d(max, 1, out));
93 
94  const int nbytes = mp_unsigned_octet_size(out);
95 
96  // Figure out how many MSBs we need to get in the
97  // most-significant byte.
98  P_CHECKA(max_bytes = calloc(nbytes, sizeof(unsigned char)));
99  MP_CHECKC(mp_to_fixlen_octets(out, max_bytes, nbytes));
100  const unsigned char mask = msb_mask(max_bytes[0]);
101 
102  // Buffer to store the pseudo-random bytes
103  P_CHECKA(buf = calloc(nbytes, sizeof(unsigned char)));
104 
105  do {
106  // Use rejection sampling to find a value strictly less than max.
107  P_CHECKC(rng_func(user_data, buf, nbytes));
108 
109  // Mask off high-order bits that we will never need.
110  P_CHECKC(rng_func(user_data, &buf[0], 1));
111  if (mask)
112  buf[0] &= mask;
113 
114  MP_CHECKC(mp_read_unsigned_octets(out, buf, nbytes));
115  } while (mp_cmp(out, max) != -1);
116 
117 cleanup:
118  if (max_bytes)
119  free(max_bytes);
120  if (buf)
121  free(buf);
122 
123  return rv;
124 }
