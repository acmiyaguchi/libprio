18 // The PrioConfig object stores "2^k-th roots of unity" modulo
19 // the prime modulus we use for all arithmetic. We use
20 // these roots to perform fast FFT-style polynomial
21 // interpolation and evaluation.
22 //
23 // In particular, we use a prime modulus p such that
24 // p = (2^k)q + 1.
25 // The roots are integers such that r^{2^k} = 1 mod p.
26 static SECStatus
27 initialize_roots(MPArray arr, const char values[], bool inverted)
28 {
29  // TODO: Read in only the number of roots of unity we need.
30  // Right now we read in all 4096 roots whether or not we use
31  // them all.
32  MP_CHECK(mp_read_radix(&arr->data[0], &values[0], 16));
33  unsigned int len = arr->len;
34  unsigned int n_chars = len * RootWidth;
35 
36  if (n_chars != sizeof(Roots)) {
37  return SECFailure;
38  }
39 
40  if (inverted) {
41  for (unsigned int i = n_chars - RootWidth, j = 1; i > 0;
42  i -= RootWidth, j++) {
43  MP_CHECK(mp_read_radix(&arr->data[j], &values[i], 16));
44  }
45  } else {
46  for (unsigned int i = RootWidth, j = 1; i < n_chars; i += RootWidth, j++) {
47  MP_CHECK(mp_read_radix(&arr->data[j], &values[i], 16));
48  }
49  }
50 
51  return SECSuccess;
52 }

61 PrioConfig
62 PrioConfig_new(int n_fields, PublicKey server_a, PublicKey server_b,
63  const unsigned char* batch_id, unsigned int batch_id_len)
64 {
65  SECStatus rv = SECSuccess;
66  PrioConfig cfg = malloc(sizeof(*cfg));
67  if (!cfg)
68  return NULL;
69 
70  cfg->batch_id = NULL;
71  cfg->batch_id_len = batch_id_len;
72  cfg->server_a_pub = server_a;
73  cfg->server_b_pub = server_b;
74  cfg->num_data_fields = n_fields;
75  cfg->n_roots = 1 << Generator2Order;
76  MP_DIGITS(&cfg->modulus) = NULL;
77  MP_DIGITS(&cfg->inv2) = NULL;
78  cfg->roots = NULL;
79  cfg->rootsInv = NULL;
80 
81  P_CHECKCB(cfg->n_roots > 1);
82  P_CHECKCB(cfg->num_data_fields <= PrioConfig_maxDataFields());
83 
84  P_CHECKA(cfg->batch_id = malloc(batch_id_len));
85  strncpy((char*)cfg->batch_id, (char*)batch_id, batch_id_len);
86 
87  MP_CHECKC(mp_init(&cfg->modulus));
88  MP_CHECKC(mp_read_radix(&cfg->modulus, Modulus, 16));
89 
90  // Compute 2^{-1} modulo M
91  MP_CHECKC(mp_init(&cfg->inv2));
92  mp_set(&cfg->inv2, 2);
93  MP_CHECKC(mp_invmod(&cfg->inv2, &cfg->modulus, &cfg->inv2));
94 
95  P_CHECKA(cfg->roots = MPArray_new(cfg->n_roots));
96  P_CHECKA(cfg->rootsInv = MPArray_new(cfg->n_roots));
97  MP_CHECKC(initialize_roots(cfg->roots, Roots, /*inverted=*/false));
98  MP_CHECKC(initialize_roots(cfg->rootsInv, Roots, /*inverted=*/true));
99 
100 cleanup:
101  if (rv != SECSuccess) {
102  PrioConfig_clear(cfg);
103  return NULL;
104  }
105 
106  return cfg;
107 }
