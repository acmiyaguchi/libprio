23 // Let the points of data_in be [x1, x2, x3, ... ].
24 // We construct the polynomial f such that
25 // (a) f(0) = random,
26 // (b) f(i) = x_i for all i >= 1,
27 // (c) degree(f)+1 is a power of two.
28 // We then evaluate f at the 2N-th roots of unity
29 // and we return these evaluations as `evals_out`
30 // and we return f(0) as `const_term`.
31 static SECStatus
32 data_polynomial_evals(const_PrioConfig cfg, const_MPArray data_in,
33  MPArray evals_out, mp_int* const_term)
34 {
35  SECStatus rv = SECSuccess;
36  const mp_int* mod = &cfg->modulus;
37  MPArray points_f = NULL;
38  MPArray poly_f = NULL;
39 
40  // Number of multiplication gates in the Valid() circuit.
41  const int mul_gates = cfg->num_data_fields;
42 
43  // Little n is the number of points on the polynomials.
44  // The constant term is randomized, so it's (mul_gates + 1).
45  const int n = mul_gates + 1;
46 
47  // Big N is n rounded up to a power of two.
48  const int N = next_power_of_two(n);
49 
50  P_CHECKA(points_f = MPArray_new(N));
51  P_CHECKA(poly_f = MPArray_new(N));
52 
53  // Set constant term f(0) to random
54  P_CHECKC(rand_int(&points_f->data[0], mod));
55  MP_CHECKC(mp_copy(&points_f->data[0], const_term));
56 
57  // Set other values of f(x)
58  for (int i = 1; i < n; i++) {
59  MP_CHECKC(mp_copy(&data_in->data[i - 1], &points_f->data[i]));
60  }
61 
62  // Interpolate through the Nth roots of unity
63  P_CHECKC(poly_fft(poly_f, points_f, cfg, true));
64 
65  // Evaluate at all 2N-th roots of unity.
66  // To do so, first resize the eval arrays and fill upper
67  // values with zeros.
68  P_CHECKC(MPArray_resize(poly_f, 2 * N));
69  P_CHECKC(MPArray_resize(evals_out, 2 * N));
70 
71  // Evaluate at the 2N-th roots of unity
72  P_CHECKC(poly_fft(evals_out, poly_f, cfg, false));
73 
74 cleanup:
75  MPArray_clear(points_f);
76  MPArray_clear(poly_f);
77 
78  return rv;
79 }

81 static SECStatus
82 share_polynomials(const_PrioConfig cfg, const_MPArray data_in,
83  PrioPacketClient pA, PrioPacketClient pB, PRG prgB)
84 {
85  SECStatus rv = SECSuccess;
86  const mp_int* mod = &cfg->modulus;
87  const_MPArray points_f = data_in;
88 
89  mp_int f0, g0;
90  MP_DIGITS(&f0) = NULL;
91  MP_DIGITS(&g0) = NULL;
92 
93  MPArray points_g = NULL;
94  MPArray evals_f_2N = NULL;
95  MPArray evals_g_2N = NULL;
96 
97  P_CHECKA(points_g = MPArray_dup(points_f));
98  P_CHECKA(evals_f_2N = MPArray_new(0));
99  P_CHECKA(evals_g_2N = MPArray_new(0));
100  MP_CHECKC(mp_init(&f0));
101  MP_CHECKC(mp_init(&g0));
102 
103  for (int i = 0; i < points_f->len; i++) {
104  // For each input value x_i, we compute x_i * (x_i-1).
105  // f(i) = x_i
106  // g(i) = x_i - 1
107  MP_CHECKC(mp_sub_d(&points_g->data[i], 1, &points_g->data[i]));
108  MP_CHECKC(mp_mod(&points_g->data[i], mod, &points_g->data[i]));
109  }
110 
111  P_CHECKC(data_polynomial_evals(cfg, points_f, evals_f_2N, &f0));
112  P_CHECKC(data_polynomial_evals(cfg, points_g, evals_g_2N, &g0));
113 
114  // The values f(0) and g(0) are set to random values.
115  // We must send to each server a share of the points
116  // f(0), g(0), and h(0) = f(0)*g(0)
117  P_CHECKC(share_int(cfg, &f0, &pA->f0_share, &pB->f0_share));
118  P_CHECKC(share_int(cfg, &g0, &pA->g0_share, &pB->g0_share));
119 
120  // Compute h(0) = f(0)*g(0).
121  MP_CHECKC(mp_mulmod(&f0, &g0, mod, &f0));
122  // Give one share of h(0) to each server.
123  P_CHECKC(share_int(cfg, &f0, &pA->h0_share, &pB->h0_share));
124 
125  // const int lenN = (evals_f_2N->len/2);
126  // P_CHECKC (MPArray_resize (pA->shares.A.h_points, lenN));
127 
128  // We need to send to the servers the evaluations of
129  // f(r) * g(r)
130  // for all 2N-th roots of unity r that are not also
131  // N-th roots of unity.
132  //
133  // For each such root r, compute h(r) = f(r)*g(r) and
134  // send a share of this value to each server.
135  int j = 0;
136  for (int i = 1; i < evals_f_2N->len; i += 2) {
137  MP_CHECKC(mp_mulmod(&evals_f_2N->data[i], &evals_g_2N->data[i], mod, &f0));
138  P_CHECKC(PRG_share_int(prgB, &pA->shares.A.h_points->data[j], &f0, cfg));
139  j++;
140  }
141 
142 cleanup:
143  MPArray_clear(evals_f_2N);
144  MPArray_clear(evals_g_2N);
145  MPArray_clear(points_g);
146  mp_clear(&f0);
147  mp_clear(&g0);
148  return rv;
149 }

151 PrioPacketClient
152 PrioPacketClient_new(const_PrioConfig cfg, PrioServerId for_server)
153 {
154  SECStatus rv = SECSuccess;
155  const int data_len = cfg->num_data_fields;
156  PrioPacketClient p = NULL;
157  p = malloc(sizeof(*p));
158  if (!p)
159  return NULL;
160 
161  p->for_server = for_server;
162  p->triple = NULL;
163  MP_DIGITS(&p->f0_share) = NULL;
164  MP_DIGITS(&p->g0_share) = NULL;
165  MP_DIGITS(&p->h0_share) = NULL;
166 
167  switch (p->for_server) {
168  case PRIO_SERVER_A:
169  p->shares.A.data_shares = NULL;
170  p->shares.A.h_points = NULL;
171  break;
172  case PRIO_SERVER_B:
173  memset(p->shares.B.seed, 0, PRG_SEED_LENGTH);
174  break;
175  default:
176  // Should never get here
177  rv = SECFailure;
178  goto cleanup;
179  }
180 
181  MP_CHECKC(mp_init(&p->f0_share));
182  MP_CHECKC(mp_init(&p->g0_share));
183  MP_CHECKC(mp_init(&p->h0_share));
184  P_CHECKA(p->triple = BeaverTriple_new());
185 
186  if (p->for_server == PRIO_SERVER_A) {
187  const int num_h_points = PrioConfig_hPoints(cfg);
188  P_CHECKA(p->shares.A.data_shares = MPArray_new(data_len));
189  P_CHECKA(p->shares.A.h_points = MPArray_new(num_h_points));
190  }
191 
192 cleanup:
193  if (rv != SECSuccess) {
194  PrioPacketClient_clear(p);
195  return NULL;
196  }
197 
198  return p;
199 }

246 bool
247 PrioPacketClient_areEqual(const_PrioPacketClient p1, const_PrioPacketClient p2)
248 {
249  if (!BeaverTriple_areEqual(p1->triple, p2->triple))
250  return false;
251  if (mp_cmp(&p1->f0_share, &p2->f0_share))
252  return false;
253  if (mp_cmp(&p1->g0_share, &p2->g0_share))
254  return false;
255  if (mp_cmp(&p1->h0_share, &p2->h0_share))
256  return false;
257  if (p1->for_server != p2->for_server)
258  return false;
259 
260  switch (p1->for_server) {
261  case PRIO_SERVER_A:
262  if (!MPArray_areEqual(p1->shares.A.data_shares, p2->shares.A.data_shares))
263  return false;
264  if (!MPArray_areEqual(p1->shares.A.h_points, p2->shares.A.h_points))
265  return false;
266  break;
267  case PRIO_SERVER_B:
268  if (memcmp(p1->shares.B.seed, p2->shares.B.seed, PRG_SEED_LENGTH))
269  return false;
270  break;
271  default:
272  // Should never get here.
273  return false;
274  }
275 
276  return true;
277 }
